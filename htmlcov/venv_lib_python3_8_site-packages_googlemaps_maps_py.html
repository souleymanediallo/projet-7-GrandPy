32class StaticMapParam(object): 

33 """Base class to handle parameters for Maps Static API.""" 

34 

35 def __init__(self): 

36 self.params = [] 

37 

38 def __str__(self): 

39 """Converts a list of parameters to the format expected by 

40 the Google Maps server. 

41 

42 :rtype: str 

43 

44 """ 

45 return convert.join_list('|', self.params)

130def static_map(client, size, 

131 center=None, zoom=None, scale=None, 

132 format=None, maptype=None, language=None, region=None, 

133 markers=None, path=None, visible=None, style=None): 

134 """ 

135 Downloads a map image from the Maps Static API. 

136 

137 See https://developers.google.com/maps/documentation/maps-static/intro 

138 for more info, including more detail for each parameter below. 

139 

140 :param size: Defines the rectangular dimensions of the map image. 

141 :type param: int or list 

142 

143 :param center: Defines the center of the map, equidistant from all edges 

144 of the map. 

145 :type center: dict or list or string 

146 

147 :param zoom: Defines the zoom level of the map, which determines the 

148 magnification level of the map. 

149 :type zoom: int 

150 

151 :param scale: Affects the number of pixels that are returned. 

152 :type scale: int 

153 

154 :param format: Defines the format of the resulting image. 

155 :type format: string 

156 

157 :param maptype: defines the type of map to construct. There are several 

158 possible maptype values, including roadmap, satellite, hybrid, 

159 and terrain. 

160 :type maptype: string 

161 

162 :param language: defines the language to use for display of labels on 

163 map tiles. 

164 :type language: string 

165 

166 :param region: defines the appropriate borders to display, based on 

167 geo-political sensitivities. 

168 :type region: string 

169 

170 :param markers: define one or more markers to attach to the image at 

171 specified locations. 

172 :type markers: StaticMapMarker 

173 

174 :param path: defines a single path of two or more connected points to 

175 overlay on the image at specified locations. 

176 :type path: StaticMapPath 

177 

178 :param visible: specifies one or more locations that should remain visible 

179 on the map, though no markers or other indicators will be displayed. 

180 :type visible: list of dict 

181 

182 :param style: defines a custom style to alter the presentation of 

183 a specific feature (roads, parks, and other features) of the map. 

184 :type style: list of dict 

185 

186 :rtype: iterator containing the raw image data, which typically can be 

187 used to save an image file locally. For example: 

188 

189 ``` 

190 f = open(local_filename, 'wb') 

191 for chunk in client.static_map(size=(400, 400), 

192 center=(52.520103, 13.404871), 

193 zoom=15): 

194 if chunk: 

195 f.write(chunk) 

196 f.close() 

197 ``` 

198 """ 

199 

200 params = {"size": convert.size(size)} 

201 

202 if not markers: 

203 if not (center or zoom is not None): 

204 raise ValueError( 

205 "both center and zoom are required" 

206 "when markers is not specifed" 

207 ) 

208 

209 if center: 

210 params["center"] = convert.latlng(center) 

211 

212 if zoom is not None: 

213 params["zoom"] = zoom 

214 

215 if scale is not None: 

216 params["scale"] = scale 

217 

218 if format: 

219 if format not in MAPS_IMAGE_FORMATS: 

220 raise ValueError("Invalid image format") 

221 params['format'] = format 

222 

223 if maptype: 

224 if maptype not in MAPS_MAP_TYPES: 

225 raise ValueError("Invalid maptype") 

226 params["maptype"] = maptype 

227 

228 if language: 

229 params["language"] = language 

230 

231 if region: 

232 params["region"] = region 

233 

234 if markers: 

235 params["markers"] = markers 

236 

237 if path: 

238 params["path"] = path 

239 

240 if visible: 

241 params["visible"] = convert.location_list(visible) 

242 

243 if style: 

244 params["style"] = convert.components(style) 

245 

246 response = client._request( 

247 "/maps/api/staticmap", 

248 params, 

249 extract_body=lambda response: response, 

250 requests_kwargs={"stream": True}, 

251 ) 

252 return response.iter_content()
